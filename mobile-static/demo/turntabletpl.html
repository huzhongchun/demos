组件名称：大转盘

点击展开详细使用说明

说明：

(string) img: //背景转盘的图片地址,

(string) arrowImg: //转盘指针的图片地址

(string) arrowWidth: //指针图片的宽度（用来将指针定位到中央）

(string) arrowHeight: //指针图片的高度（用来将指针定位到中央）

(string) arrowTop: //指针距离上面大抓盘的高度，实现指针居中的位置

(string) width: //转盘的宽度

(string) len: //扇形数目

(string) random: //是否用默认得随机方法

(string) prizeArea: //中奖信息的对象结构为 '10': { 'area': [1,3], 'message': '满100减10元现金券' }, '40': { 'area': [2,5], 'message': '满300减40元现金券' }, '70': { 'area': [4,7], 'message': '满500减70元现金券' }, '0': { 'area': [6], 'message': '很遗憾哦~' }

(string) element: //大转盘外层的元素

(string) randomChance: { "1": 100 }: //默认的随机方法对应奖项和概率

(function) success: //成功回调函数，参数为startRotate传得参数

(function) $pluginArea.startRotate('0', 'zhong'); //调用开始的方法

无

Javascript调用
//初始化 var turntableData = { '10': { 'area': [1,3], 'message': '满100减10元现金券' }, '40': { 'area': [2,5], 'message': '满300减40元现金券' }, '70': { 'area': [4,7], 'message': '满500减70元现金券' }, '0': { 'area': [6], 'message': '很遗憾哦~' } } var turntableClass = Jumei.getModule('ui.turntable'); var $pluginArea = new turntableClass({ img: 'http://images.jumei.com/mobile/act/activities/2014_12/1219_queren/zhuanpan2.png', arrowImg: 'http://images.jumei.com/mobile/act/activities/2014_12/1219_queren/arrow.png', arrowHeight: 155, arrowWidth: 109, random: false, prizeArea: turntableData, element:'.dazhuanpan', success: function(data, rewardinfo) { console.dir(data); console.dir(rewardinfo); console.dir(turntableData[data].message); } }); //出发大转盘 转动 $('#turntable .turntable-btn').on('click',function(e){ $pluginArea.startRotate('0', 'zhong'); });
